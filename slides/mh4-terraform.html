Core Implementation

Usage Plan + API Key

📄 module/GeekBrain_Backend/main.tf 
# ═══════════════════════════════════════
# API Gateway — Usage Plan + API Key (W5 MH4)
# ═══════════════════════════════════════

resource "aws_api_gateway_usage_plan" "chat" {
  name = "${var.project}-chat-usage-plan"

  api_stages {
    api_id = aws_api_gateway_rest_api.main.id
    stage  = aws_api_gateway_stage.prod.stage_name
  }

  throttle_settings {
    rate_limit  = var.api_throttle_rate_limit   # 10 req/s
    burst_limit = var.api_throttle_burst_limit  # 20 burst
  }

  quota_settings {
    limit  = var.api_quota_limit  # 1000/day
    period = "DAY"
  }

  tags = var.tags
}

resource "aws_api_gateway_api_key" "chat" {
  name    = "${var.project}-chat-api-key"
  enabled = true
  tags    = var.tags
}

resource "aws_api_gateway_usage_plan_key" "chat" {
  key_id        = aws_api_gateway_api_key.chat.id
  key_type      = "API_KEY"
  usage_plan_id = aws_api_gateway_usage_plan.chat.id
}
Authentication

API Key Required + Variables

📄 module/GeekBrain_Backend/main.tf 1 dòng thay đổi 
resource "aws_api_gateway_method" "chat_post" {
  rest_api_id      = aws_api_gateway_rest_api.main.id
  resource_id      = aws_api_gateway_resource.chat.id
  http_method      = "POST"
  authorization    = "NONE"
  api_key_required = true   # ← W5 MH4: bật API Key authentication
}
📄 module/GeekBrain_Backend/variables.tf 
variable "api_throttle_rate_limit" {
  description = "API Gateway throttle rate limit (requests/second)"
  type        = number
  default     = 10
}

variable "api_throttle_burst_limit" {
  description = "API Gateway throttle burst limit"
  type        = number
  default     = 20
}

variable "api_quota_limit" {
  description = "API Gateway daily quota limit"
  type        = number
  default     = 1000
}
